        private void BtnConectar_Click(object sender, EventArgs e)
        {
            sServidor = TbServidor.Text;
            sUsuario = TbUsuario.Text;
            sPassword = TbPasword.Text;

            sql = new DBSqlServer(TbServidor.Text, TbUsuario.Text, TbPasword.Text);

            if (sql.AbrirConexion())
            {
                MessageBox.Show("Conexion realizada...");
                TbConsola.Enabled = true;
                BtnCrearTabla.Enabled = true;
                BtnEjecutar.Enabled = true;
                TvTblas.Nodes.Clear();

                try
                {
                    TvServidores.Nodes.Clear();
                    DataTable Tabla = sql.VerTodasLasDB();

        private void TvServidores_NodeMouseDoubleClick(object sender, TreeNodeMouseClickEventArgs e)

            {
                DB = TvServidores.SelectedNode.Text;
                TvTblas.Nodes.Clear();
                String[] Elementos;
                DataTable Elemento;
                TreeNode TabActual;

                T = sql.VerTablas(DB);
                foreach (DataRow dr in T.Rows)
                {
                    TvTblas.Nodes.Add(dr[0].ToString());

                    TvTblas.Nodes[TvTblas.Nodes.Count - 1].Nodes.Add("Elementos");
                    TvTblas.Nodes[TvTblas.Nodes.Count - 1].Nodes.Add("Keys");
                    TvTblas.Nodes[TvTblas.Nodes.Count - 1].Nodes.Add("Constrains");

                    TabActual = TvTblas.Nodes[TvTblas.Nodes.Count - 1];

                    //------------------------------------------Elementos----------------------------------------------------

                    Elementos = sql.VerElementos(DB, dr[0].ToString());
                    foreach (String e1 in Elementos)
                    {
                        TvTblas.Nodes[TabActual.Index].Nodes[0].Nodes.Add(e1);
                    }

                    //------------------------------------------Keys----------------------------------------------------

                    Elementos = sql.VerKeys(DB, dr[0].ToString());
                    foreach (String e2 in Elementos)
                    {
                        TvTblas.Nodes[TabActual.Index].Nodes[1].Nodes.Add(e2);
                    }

                    //------------------------------------------Constrains----------------------------------------------------

                    Elemento = sql.VerConstrains(DB, dr[0].ToString());
                    foreach (DataRow e3 in Elemento.Rows)
                    {
                        TvTblas.Nodes[TabActual.Index].Nodes[2].Nodes.Add(e3[1].ToString());
                    }

                }

                sql.CambiarDB(DB);
                MessageBox.Show($"Base de datos en uso: {sql.VerDB()}");

            }
            catch (Exception ex)
            {
                sql.sLastError = ex.Message;
                MessageBox.Show($"Ha ocurrido un error:\n {sql.sLastError}", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }

        private void BtnCrear_Click(object sender, EventArgs e)
        {
            if (DGBTabla.RowCount >= 1)
            {
                string sentencias = "";
                int cont = 0;
                foreach (string a in elementos)
                {
                    if (cont == 0)
                    {
                        sentencias = a;
                    }
                    else
                    {
                        sentencias += $", {a}";
                    }
                    cont++;
                }

                try
                {
                    if(!sql.Ejecutar($"Create table {tnombre} ( {sentencias} );"))
                    {

                            MessageBox.Show($"Ha ocurrido un error:\n {sql.sLastError}", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                    }
                    else
                    {
                        MessageBox.Show($"La tabla {tnombre} fue creada correctamente");
                    }

                }
                catch
                {
                    MessageBox.Show($"Ha ocurrido un error:\n {sql.sLastError}", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }

                DGBTabla.Rows.Clear();
            }
            else
            {
                MessageBox.Show($"Por favor agrega al menos una linea a la tabla", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }


        }

                dnombre = TbNombreDato.Text;
                tipo = CbTipoDato.Text;
                nulos = CkbNullos.Checked;
                autin = CkbAutoIncrementable.Checked;
                pk = CkbPrimaryKey.Checked;

                if (nulos) { linea = $"{dnombre} {tipo}"; } else { linea += " not null"; }

                DGBTabla.Rows.Add( dnombre, tipo, nulos, autin, pk );

                Default();
                BtnCrear.Enabled = true;
            }
            else
            {
                MessageBox.Show($"Por favor llena los requisitos mínimos", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }

    public class DBSqlServer
    {
        SqlConnection conexion = null;
        public String sLastError = "";
        public DataTable Tabla = new DataTable();

        public DBSqlServer (String sServer, String sUsuario, String sContraseña)
        {
            conexion = new SqlConnection($"Server={sServer}; Database=master; User Id={sUsuario}; Password={sContraseña};");
        }

        //------------------------------------------------------------------------------------Acciones----------------------------------------------------------------------------------------------

        public Boolean AbrirConexion()
        {
            Boolean bAllOk = true;
            try
            {
                conexion.Open();
            }
            catch(Exception ex) 
            {
                sLastError = ex.Message;
                bAllOk = false;
            }
            return bAllOk;
        }

        public Boolean ConexionAbierta()
        {
            Boolean bAllOk = true;
            try
            {
                bAllOk = conexion.State == System.Data.ConnectionState.Open ? true : false;
            }
            catch (Exception ex)
            {
                sLastError = ex.Message;
                bAllOk = false;
            }
            return bAllOk;
        }

        public Boolean Ejecutar(String sCmd)
        {
            Boolean bAllOk = true;

            String[] sValida = sCmd.Split(' ');

            try
            {

                using (SqlCommand Cmd = new SqlCommand(sCmd, conexion))
                {
                    Cmd.ExecuteNonQuery();
                    if (sValida[0] == "select" || sValida[0] == "Select") 
                    { 
                        Tabla.Load(Cmd.ExecuteReader()); 
                    }                    
                }
                conexion.Close();

            }
            catch (Exception ex)
            {
                sLastError = ex.Message;
                bAllOk = false;
            }


            return bAllOk;
        }

        public DataTable EjecutarTabla(String sCmd)
        {
            DataTable Tabla = new DataTable();
            try
            {

                using (SqlCommand Cmd = new SqlCommand(sCmd, conexion))
                {
                    Cmd.ExecuteNonQuery();
                    Tabla.Load(Cmd.ExecuteReader());
                }

            }
            catch (Exception ex)
            {
                sLastError = ex.Message;
            }
            return Tabla;
        }

        public void CerrarConexion()
        {
            conexion.Close();
        }

            {
                sLastError = ex.Message;

        {
            Boolean bOk = false;

            }

        public String[] VerElementos(String DB, String sNombreTabla)
        {
            DataTable Tabla = new DataTable();
            String[] Lineas = new String[1];
            try
            {
                string Query = $"Use {DB} select "+
                    " COLUMN_NAME" +
                    ", DATA_TYPE"+
                    ", CHARACTER_MAXIMUM_LENGTH" +
                    ", NUMERIC_PRECISION" +
                    ", NUMERIC_SCALE" +
                    ", IS_NULLABLE " +
                    $"from INFORMATION_SCHEMA.COLUMNS where TABLE_NAME = '{sNombreTabla}'";

                using (SqlCommand Cmd = new SqlCommand(Query, conexion))
                {
                    Cmd.ExecuteNonQuery();
                    Tabla.Load(Cmd.ExecuteReader());
                }
                Lineas = new String[Tabla.Rows.Count];

                String[] Drow = VerKeys(DB, sNombreTabla);
                String[] Drow2;

                int Cont = 0;
                Boolean EsPk = false;
                foreach (DataRow Dr in Tabla.Rows)
                {
                    foreach(string s in Drow)
                    {
                        Drow2 = (s.Split('_'));
                        if (Drow2[1].ToString() == Dr[0].ToString())
                        {
                            Lineas[Cont] = Drow2[0] + $" {Dr[0]} ({Drow2[0]}, {Dr[1]}";
                            EsPk = true;
                            break;
                        }
                    }

                    if (EsPk == false) { Lineas[Cont] += $"{Dr[0]} ({Dr[1]}"; }
                    if (Dr[2].ToString() != "") { Lineas[Cont] += $"({Dr[2]})"; }

                    if (Dr[3].ToString() != "" && (Dr[4].ToString() != "" && Dr[4].ToString() != "0")) { Lineas[Cont] += $"({Dr[3]},{Dr[4]})"; }

                    if (Dr[5].ToString() == "NO") { Lineas[Cont] += ", not null"; } else { Lineas[Cont] += ", null"; }
                    Lineas[Cont] += ")";

                    Cont++;
                    EsPk = false;
                }

            }
            catch (Exception ex)
            {
                sLastError = ex.Message;
            }

            return Lineas;
        }

        public String[] VerKeys(String DB, String sNombreTabla)
        {
            DataTable Tabla = new DataTable();
            String[] Lipe = new string[0];

            try
            {
                string Query = $"Use {DB} select " +
                    " CONSTRAINT_NAME " +
                    $" from INFORMATION_SCHEMA.KEY_COLUMN_USAGE  where TABLE_NAME = '{sNombreTabla}'";

                using (SqlCommand Cmd = new SqlCommand(Query, conexion))
                {
                    Cmd.ExecuteNonQuery();
                    Tabla.Load(Cmd.ExecuteReader());
                }
                Lipe = new string[Tabla.Rows.Count];

                int Cont = 0;
                foreach (DataRow Der in Tabla.Rows)
                {
                    Lipe[Cont] = Der[0].ToString();

                    Cont++;
                }

            }
            catch (Exception ex)
            {
                sLastError = ex.Message;
            }

            return Lipe;
        }
